    from puresnmp import bulkwalk
    ip = '127.0.0.1'
    community = 'private'
    oids = [
        '1.3.6.1.2.1.6',
        '1.3.6.1.2.1.7',

    result = bulkwalk(ip, community, oids)
    for k, v in result:
